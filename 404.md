<script>
(function() {
  // Get the current path
  var currentPath = window.location.pathname;
  var currentHash = window.location.hash;
  var currentSearch = window.location.search;
  
  // Try lowercase version first
  var lowerPath = currentPath.toLowerCase();
  if (currentPath !== lowerPath) {
    // Redirect to lowercase version
    window.location.href = lowerPath + currentSearch + currentHash;
    return;
  }
  
  // If already lowercase, try appending trailing slash
  if (!currentPath.endsWith('/') && !currentPath.match(/\.[a-zA-Z0-9]+$/)) {
    // Only add trailing slash if it doesn't look like a file (no extension)
    window.location.href = currentPath + '/' + currentSearch + currentHash;
    return;
  }
  
  // If we get here, both redirects have been tried or wouldn't help
  // Show the friendly message
  document.getElementById('redirect-message').style.display = 'none';
  document.getElementById('not-found-message').style.display = 'block';
})();
</script>
